```java
public class Main {
    public static void main(String[] args) {
        SutdaCard card1 = new SutdaCard(3, false);
        SutdaCard card2 = new SutdaCard();
        System.out.println(card1.info());
        System.out.println(card2.info());
    }
}

class SutdaCard {
    int num;
    boolean isKwang;

    SutdaCard(int inputNum, boolean inputKwang) {
        num = inputNum;
        isKwang = inputKwang;
    }

    SutdaCard() {
        num = (int) (Math.random() * 10);  //정수를 따로 지정하지 않으면 1~10 랜덤값 받음.
        isKwang = true;   //광 여부는 따로 지정하지 않으면 true로 설정됨.
    }

    String info() {
        String Allinfo;
        Allinfo = Integer.toString(num);
        if (isKwang == true) {
            Allinfo += "K";
        }
        return Allinfo;
    }
}
```

### 6-5

```java
public class Main {
    public static void main(String[] args) {
        Student s = new Student("홍길동", 1, 1, 100, 60, 76);
        System.out.println(s.info());
    }
}

class Student {

    String name;  //학생이름
    int ban;      //반
    int no;       //번호
    int kor;      //국어점수
    int eng;      //영어점수
    int math;     //수학점수

    Student(String setName, int setBan, int setNo, int setKor, int setEng, int setMath) {
        name = setName;
        ban = setBan;
        no = setNo;
        kor = setKor;
        eng = setEng;
        math = setMath;
    }

    int getTotal() {
        return kor + eng + math;
    }

    float getAverage() {
        final int SUBJECTNUM = 3;
        float avr = getTotal() / (float) SUBJECTNUM;
        return Math.round(avr * 10) / 10.0F;
    }

    String info() {
        return name + "," + ban + "," + no + "," + kor + "," + eng + "," + math + "," + getTotal() + "," + getAverage();
    }
}
```
