import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int N = sc.nextInt();
        int X = sc.nextInt();
        int[] myArray = new int[N];
        for (int i = 0; i < N; i++) {
            myArray[i] = sc.nextInt();
        }
        sc.close();
        for (int i = 0; i < N; i++) {
            if (myArray[i] < X) {
                System.out.printf("%d ", myArray[i]);
            }
        }

    }
}
